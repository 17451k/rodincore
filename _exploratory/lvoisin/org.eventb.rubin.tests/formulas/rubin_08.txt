"VALID, Rubin Exercise 8.A(1), p173":
!x (Ax & Bx -> Cx),
#x (Bx & ~Cx)
|- #x ~Ax;

"VALID, Rubin Exercise 8.A(2), p173":
!x (Px -> ~Qx | ~Rx),
#x (Qx & Rx)
|- #x ~Px;

"VALID, Rubin Exercise 8.A(3), p173":
!x (Ax -> Cx),
!x (Bx -> Dx),
#x (Ax & Bx)
|- #x (Cx & Dx);

"VALID, Rubin Exercise 8.A(4), p173":
!x (Px -> Qx),
#x (Rx & ~Qx),
!x (Rx -> Px | Sx)
|- #x (Rx & Sx);

"VALID, Rubin Exercise 8.A(5), p173":
!x (~Ax -> Bx),
!x (~Bx | Cx),
#x ~Cx & #x Cx
|- Ca | #x Ax;

"VALID, Rubin Exercise 8.A(6), p173":
!x (Px | Qx | Rx),
!x (Px -> ~Sx),
~Ra & Sa
|- #x (~Px & Qx);

"VALID, Rubin Exercise 8.A(7), p173":
!x !y (Ax & By -> Cxy),
#x (Ax & Cax),
#x (Bx & ~Cax)
|- #x ~Ax;

"VALID, Rubin Exercise 8.A(8), p173":
#y !x Rxy
|- !x #y Rxy;

"VALID, Rubin Exercise 8.A(9), p173":
#x Ax,
#x Bx,
!x (Ax -> !y (By -> Cxy))
|- #x #y Cxy;

"VALID, Rubin Exercise 8.A(10), p173":
#x (Px & !y (Qy -> Rxy)),
!x !y (Px & Rxy -> Sy),
!x (Sx -> Qx)
|- !x (Qx <-> Sx);

"VALID, Rubin Exercise 8.A(11), p173":
!x !y (Dx & Ey -> Fxy),
!x !y (Dx & Fxy -> Gy)
|- #x Dx -> !x (Ex -> Gx);

"VALID, Rubin Exercise 8.A(12), p173":
!x !y (Jx & ~Ky -> Lxy),
#x (Jx & !y (My -> ~Lxy))
|- !x (Mx -> Kx);

"VALID, Rubin Exercise 8.A(13), p173":
#x (Qx & !y (Py -> Rxy)),
!x (Px -> !y (Qy & Sy -> ~Ryx))
|- #x Px -> #x ~Sx;

"VALID, Rubin Exercise 8.A(14), p173":
#x (Px & !y (Py & Rxy -> Qya)),
#x (Px & ~Qxa),
#x (~Px & Qxa)
|- #x #y (Px & Py & ~Rxy);

"VALID, Rubin Exercise 8.A(15), p174":

"VALID, Rubin Exercise 8.A(16), p174":
